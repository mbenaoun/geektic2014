CREATE TABLE GEEK 
( 
ID_GEEK INTEGER IDENTITY,
NOM_GEEK VARCHAR(20),
PRENOM_GEEK VARCHAR(20),
AGE_GEEK INTEGER,
SEXE_GEEK VARCHAR(5),
MAIL_GEEK VARCHAR(30),
PATHAVATAR_GEEK VARCHAR(50)
);

CREATE TABLE CENTRES_INTERET
(
ID_INTERET INTEGER IDENTITY,
INTERET VARCHAR(30)
);

CREATE TABLE GEEK_CENTRES_INTERET
(
ID_GEEK INTEGER, 
ID_INTERET INTEGER
);


ALTER TABLE GEEK
ADD CONSTRAINT PK_GEEK PRIMARY KEY (ID_GEEK);

ALTER TABLE GEEK
ADD CONSTRAINT GEEK_SEXE_CK CHECK(SEXE_GEEK IN ('F', 'M'));

ALTER TABLE CENTRES_INTERET
ADD CONSTRAINT PK_CENTRES_INTERET PRIMARY KEY (ID_INTERET);

ALTER TABLE GEEK_CENTRES_INTERET
ADD CONSTRAINT PK_GEEK_CENTRES_INTERET PRIMARY KEY (ID_GEEK, ID_INTERET);

ALTER TABLE GEEK_CENTRES_INTERET
ADD CONSTRAINT FK_GCI_IDGEEK FOREIGN KEY (ID_GEEK) REFERENCES PUBLIC.GEEK (ID_GEEK);

ALTER TABLE GEEK_CENTRES_INTERET
ADD CONSTRAINT FK_GCI_IDINTERET FOREIGN KEY (ID_INTERET) REFERENCES PUBLIC.CENTRES_INTERET (ID_INTERET);
